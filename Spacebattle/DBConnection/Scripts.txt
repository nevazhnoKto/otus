CREATE DATABASE Avito;

-- Создание таблицы "users" с первичным ключом "user_id"
CREATE TABLE users (
    user_id SERIAL PRIMARY KEY,
    username VARCHAR(50) NOT NULL,
    email VARCHAR(100) NOT NULL
);

-- Создание таблицы "posts" с первичным ключом "post_id" и внешним ключом "user_id" связанным с таблицей "users"
CREATE TABLE posts (
    post_id SERIAL PRIMARY KEY,
    user_id INTEGER REFERENCES users(user_id),
    title VARCHAR(100) NOT NULL,
    content TEXT
);

-- Создание таблицы "comments" с первичным ключом "comment_id" и внешним ключом "post_id" связанным с таблицей "posts"
CREATE TABLE comments (
    comment_id SERIAL PRIMARY KEY,
    post_id INTEGER REFERENCES posts(post_id),
    user_id INTEGER REFERENCES users(user_id),
    comment_text TEXT
);
